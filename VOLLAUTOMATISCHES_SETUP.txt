export function loadEnvironment() {
	const platform = process.platform;
	let envFile;

	if (platform === 'win32') {
		envFile = '.env.development';
	} else if (platform === 'darwin') {
		envFile = '.env.testing';
	} else {
		envFile = '.env.production';
	}

	const envPath = resolve(__dirname, '../../', envFile);

	if (!fs.existsSync(envPath)) {
		console.error(`âŒ Environment file not found: ${envPath}`);
		process.exit(1);
	}

	const result = dotenv.config({ path: envPath });

	if (result.error) {
		console.error('âŒ Error loading environment file:', result.error);
		process.exit(1);
	}

	console.log(`âœ… Environment loaded: ${process.env.NODE_ENV} (${platform})`);
	console.log({
		envFile,
		config: {
			port: process.env.PORT,
			apiBaseUrl: process.env.VITE_API_BASE_URL,
			logLevelServer: process.env.LOG_LEVEL_SERVER,
			logLevelClient: process.env.VITE_LOG_LEVEL_CLIENT,
			enableSourceMaps: process.env.ENABLE_SOURCE_MAPS === 'true',
			jwtExpiry: process.env.JWT_EXPIRY
		}
	});

	return result.parsed;
}

import fs from 'fs';
import path from 'path';
import { fileURLToPath } from 'url';

const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);

const CONFIG = {
	logsDir: path.resolve(__dirname, '../../logs/server'),
	maxFileSize: 100 * 1024 * 1024,
	maxFiles: 10,
	maxAge: 30 * 24 * 60 * 60 * 1000,
	encoding: 'utf8'
};

function ensureLogsDirectory() {
	if (!fs.existsSync(CONFIG.logsDir)) {
		fs.mkdirSync(CONFIG.logsDir, { recursive: true });
	}
}

function getCurrentLogFile() {
	const date = new Date().toISOString().split('T')[0];
	return path.join(CONFIG.logsDir, `server-${date}.log`);
}

function getFileSize(filePath) {
	try {
		const stats = fs.statSync(filePath);
		return stats.size;
	} catch {
		return 0;
	}
}

function getFileAge(filePath) {
	try {
		const stats = fs.statSync(filePath);
		return Date.now() - stats.mtime.getTime();
	} catch {
		return 0;
	}
}

function rotateLogFile(filePath) {
	try {
		const timestamp = new Date().toISOString().replace(/[:.]/g, '-');
		const newPath = filePath.replace('.log', `-${timestamp}.log`);
		fs.renameSync(filePath, newPath);
	} catch (err) {
		console.error('Failed to rotate log file:', err);
	}
}

function cleanOldLogs() {
	try {
		const files = fs.readdirSync(CONFIG.logsDir)
			.filter(f => f.startsWith('server-') && f.endsWith('.log'))
			.map(f => ({
				name: f,
				path: path.join(CONFIG.logsDir, f),
				mtime: fs.statSync(path.join(CONFIG.logsDir, f)).mtime.getTime()
			}))
			.sort((a, b) => b.mtime - a.mtime);

		files.forEach(file => {
			const age = getFileAge(file.path);
			if (age > CONFIG.maxAge) {
				fs.unlinkSync(file.path);
			}
		});

		if (files.length > CONFIG.maxFiles) {
			files.slice(CONFIG.maxFiles).forEach(file => {
				fs.unlinkSync(file.path);
			});
		}
	} catch (err) {
		console.error('Failed to clean old logs:', err);
	}
}

function writeToFile(message) {
	try {
		ensureLogsDirectory();
		const logFile = getCurrentLogFile();
		const fileSize = getFileSize(logFile);

		if (fileSize >= CONFIG.maxFileSize) {
			rotateLogFile(logFile);
		}

		fs.appendFileSync(logFile, message + '\n', { encoding: CONFIG.encoding });
		cleanOldLogs();
	} catch (err) {
		console.error('Failed to write to log file:', err);
	}
}

const levels = {
	debug: { icon: 'ğŸ›', color: '\x1b[36m', name: 'DEBUG' },
	info: { icon: 'â„¹ï¸', color: '\x1b[34m', name: 'INFO' },
	success: { icon: 'âœ…', color: '\x1b[32m', name: 'SUCCESS' },
	warn: { icon: 'âš ï¸', color: '\x1b[33m', name: 'WARN' },
	error: { icon: 'âŒ', color: '\x1b[31m', name: 'ERROR' },
	fatal: { icon: 'ğŸ’€', color: '\x1b[35m', name: 'FATAL' }
};

function log(level, source, message, data = null) {
	const timestamp = new Date().toISOString();
	const levelInfo = levels[level] || levels.info;
	const logLevel = process.env.LOG_LEVEL_SERVER || 'info';
	const logLevelPriority = ['debug', 'info', 'success', 'warn', 'error', 'fatal'];

	if (logLevelPriority.indexOf(level) < logLevelPriority.indexOf(logLevel)) {
		return;
	}

	const consoleMessage = `${levelInfo.icon} [${timestamp}] ${levelInfo.color}[${levelInfo.name}]\x1b[0m [${source}] ${message}`;
	console.log(consoleMessage);
	if (data) {
		console.log(data);
	}

	const fileMessage = `${levelInfo.icon} [${timestamp}] [${levelInfo.name}] [${source}] ${message}${data ? ' ' + JSON.stringify(data) : ''}`;
	writeToFile(fileMessage);
}

export const logger = {
	debug: (source, message, data) => log('debug', source, message, data),
	info: (source, message, data) => log('info', source, message, data),
	success: (source, message, data) => log('success', source, message, data),
	warn: (source, message, data) => log('warn', source, message, data),
	error: (source, message, data) => log('error', source, message, data),
	fatal: (source, message, data) => log('fatal', source, message, data),
	http: (method, path, status, duration) => {
		const level = status >= 500 ? 'error' : status >= 400 ? 'warn' : 'success';
		log(level, 'HTTP', `${method} ${path} ${status} - ${duration}ms`);
	},
	api: (endpoint, method, status, data) => {
		const level = status >= 400 ? 'error' : 'success';
		log(level, 'API', `${method} ${endpoint} - ${status}`, data);
	},
	auth: (action, details, data) => {
		log('info', 'Auth', `${action}: ${details}`, data);
	},
	getConfig: () => CONFIG,
	cleanup: cleanOldLogs

async function sendToServer(level, source, message, data) {
	try {
		const apiBaseUrl = import.meta.env.VITE_API_BASE_URL || 'http://localhost:5000';
		await fetch(`${apiBaseUrl}/logs/client`, {
			method: 'POST',
			headers: { 'Content-Type': 'application/json' },
			body: JSON.stringify({
				level,
				source,
				message,
				data,
				timestamp: new Date().toISOString(),
				userAgent: navigator.userAgent,
				url: window.location.href
			})
		});
	} catch (err) {
		console.error('Failed to send log to server:', err);
	}
}

function log(level, source, message, data = null) {
	const timestamp = new Date().toISOString();
	const levelInfo = levels[level] || levels.info;
	const logLevel = import.meta.env.VITE_LOG_LEVEL_CLIENT || 'info';
	const logLevelPriority = ['debug', 'info', 'success', 'warn', 'error', 'fatal'];

	if (logLevelPriority.indexOf(level) < logLevelPriority.indexOf(logLevel)) {
		return;
	}

	const styles = [
		`color: ${levelInfo.color}`,
		'font-weight: bold'
	].join(';');

	console.log(
		`${levelInfo.icon} [${timestamp}] %c[${levelInfo.name}]%c [${source}] ${message}`,
		styles,
		'',
		data || ''
	);

	sendToServer(level, source, message, data);
}

	error: (source, message, data) => log('error', source, message, data),

	},
